const DataTypes = require("sequelize");
const dbConnect = require("../config/elephantSql/elephantConnect");

const sequelize = dbConnect();

/**
 * @description User model
 * @name User
 */
/**
 * @typedef user
 * @property {email} - User's email
 * @property {uuid} - User's unique identifier
 * @property {role} - User's role - determines level of access
 * @property {first_name} - User's first name
 * @property {last_name} - User's last name
 * @property {password} - Hash of user's password
 */

const User = sequelize.define("users", {
  email: {
    type: DataTypes.STRING,
    unique: true,
  },
  uuid: {
    type: DataTypes.UUID,
    defaultValue: DataTypes.UUIDV4,
    unique: true,
  },
  role: {
    type: DataTypes.INTEGER,
  },
  first_name: {
    type: DataTypes.STRING,
  },
  last_name: {
    type: DataTypes.STRING,
  },
  password: {
    type: DataTypes.STRING,
  },
});

module.exports = User;
